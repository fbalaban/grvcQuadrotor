184 void GazeboQuadrotorSimpleController::StateCallback(const nav_msgs::OdometryConstPtr& state)
185 {
186  math::Vector3 velocity1(velocity);
187 
188  if (imu_topic_.empty()) {
189  pose.pos.Set(state->pose.pose.position.x, state->pose.pose.position.y, state->pose.pose.position.z);
190  pose.rot.Set(state->pose.pose.orientation.w, state->pose.pose.orientation.x, state->pose.pose.orientation.y, state->pose.pose.orientation.z);
191  euler = pose.rot.GetAsEuler();
192  angular_velocity.Set(state->twist.twist.angular.x, state->twist.twist.angular.y, state->twist.twist.angular.z);
193  }
194 
195  velocity.Set(state->twist.twist.linear.x, state->twist.twist.linear.y, state->twist.twist.linear.z);
196 
197  // calculate acceleration
198  double dt = !state_stamp.isZero() ? (state->header.stamp - state_stamp).toSec() : 0.0;
199  state_stamp = state->header.stamp;
200  if (dt > 0.0) {
201  acceleration = (velocity - velocity1) / dt;
202  } else {
203  acceleration.Set();
204  }

382 double GazeboQuadrotorSimpleController::PIDController::update(double new_input, double x, double dx, double dt)
383 {
384  // limit command
385  if (limit > 0.0 && fabs(new_input) > limit) new_input = (new_input < 0 ? -1.0 : 1.0) * limit;
386 
387  // filter command
388  if (dt + time_constant > 0.0) {
389  dinput = (new_input - input) / (dt + time_constant);
390  input = (dt * new_input + time_constant * input) / (dt + time_constant);
391  }
392 
393  // update proportional, differential and integral errors
394  p = input - x;
395  d = dinput - dx;
396  i = i + dt * p;
397 
398  // update control output
399  output = gain_p * p + gain_d * d + gain_i * i;
400  return output;
401 }
