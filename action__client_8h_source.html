81  template<class Goal_, class FeedBack_>
82  ActionClient<Goal_, FeedBack_>::ActionClient(const char* _node_name, const char* _topic_base, int _argc, char** _argv) {
83  std::string ns = std::string(_topic_base) + "/";
84  // Create publishers
85  goal_pub_ = new Publisher(_node_name, (ns + "goal").c_str(), _argc, _argv);
86  abort_pub_ = new Publisher(_node_name, (ns + "abort").c_str(), _argc, _argv);
87  // Goal state subscription
88  state_sub_ = new Subscriber<std::string>(_node_name, (ns + "state").c_str(), _argc, _argv, [this](const std::string& _state) {
89  if(_state == "pending")
90  cur_state_ = GoalState::pending;
91  else if(_state == "accepted")
92  cur_state_ = GoalState::accepted;
93  else if(_state == "rejected")
94  cur_state_ = GoalState::rejected;
95  else if(_state =="success")
96  cur_state_ = GoalState::success;
97  else if (_state == "fail")
98  cur_state_ = GoalState::fail;
99  else if (_state == "cancelled")
100  cur_state_ = GoalState::cancelled;
101  if (cur_state_ == GoalState::success || cur_state_ == GoalState::fail) {
102  goal_cb_(cur_state_);
103  }
104  });
105  // Feedback subscription
106  fb_sub_ = new Subscriber<FeedBack_>(_node_name, (ns + "feedback").c_str(), _argc, _argv, [this](const FeedBack_& _fb) {
107  feedback_cb_(_fb);
108  });
109  }
