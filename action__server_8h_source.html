67  template<class Goal_, class FeedBack_>
68  ActionServer<Goal_, FeedBack_>::ActionServer(const char* _node_name, const char* _topic_base, int _argc, char** _argv) {
69  std::string ns = std::string(_topic_base) + "/";
70  // Create publishers
71  fb_pub_ = new Publisher(_node_name, (ns + "feedback").c_str(), _argc, _argv);
72  state_pub_ = new Publisher(_node_name, (ns + "state").c_str(), _argc, _argv);
73  // Goal subscription
74  goal_sub_ = new Subscriber<Goal_>(_node_name, (ns + "goal").c_str(), _argc, _argv, [this](const Goal_& _goal) {
75  if (goal_cb_(_goal)) {
76  state_pub_->publish("accepted");
77  }
78  else {
79  state_pub_->publish("rejected");
80  }
81  });
82  // Feedback subscription
83  abort_sub_ = new Subscriber<void>(_node_name, (ns + "abort").c_str(), _argc, _argv, [this]() {
84  abort_cb_();
85  state_pub_->publish("cancelled");
86  });
87  }
